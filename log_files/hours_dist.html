google.setOnLoadCallback(drawHoursDistribution); function drawHoursDistribution() { var data = new google.visualization.DataTable(); data.addColumn('string', 'Hour of the day'); data.addColumn('number', 'Number of failed logins'); var chart = new google.visualization.ColumnChart(document.getElementById('hours_dist_div')); chart.draw(data, {width: 800, height: 550, title: 'Failed logins by hour of the day'}); }
